def update(request, pk):
    update = TdDiscrepancyrecordform.objects.get(transaction_number = pk)
    form = ShipmentForm(instance=update)
  
    if request.method == 'POST':
        form = ShipmentForm(request.POST, instance=update)
        if form.is_valid():
            form.save()    
            return redirect('Retrieve')
